% Setting the paragraph indentation to 0
% | setlength - Allows you to define various lengths in LaTeX
% | parindent - Refers to the indentation of paragraphs
\setlength{\parindent}{0pt}

% Adding a bibliography resource for BibLaTeX
% | addbibresource - Command to add a .bib file as a resource for citations
\addbibresource{Keen.bib}

% Configuring the main font for the document
% | setmainfont - Defines the main font for the document using the fontspec package
% | Path - Directory location of the font files
% | BoldFont, ItalicFont, etc. - Different styles of the font
\setmainfont{MinionPro-Regular.otf}[
  Path = fonts/,
  BoldFont = MinionPro-Bold.otf,
  ItalicFont = MinionPro-It.otf,
  BoldItalicFont = MinionPro-BoldIt.otf,
]

% Configuring the monospace font (often used for code or typewriter text)
% | setmonofont - Defines the monospace font for the document
% | Scale - Adjusts the size of the font relative to the main font
\setmonofont{DejaVuSansMNerdFontMono-Regular.ttf}[
  Path = fonts/,
  BoldFont = DejaVuSansMNerdFontMono-Bold.ttf,
  ItalicFont = DejaVuSansMNerdFontMono-Oblique.ttf,
  Scale = 0.85
]

% Setting up hyperlink properties
% | hypersetup - Command to set properties for hyperlinks in the document
% | colorlinks - Colors the text of the links instead of surrounding it with colored boxes
% | linkcolor, citecolor, etc. - Define the color for various types of links
% | pdftitle - Specifies the title of the PDF when it's created
\hypersetup{
    colorlinks=true,
    linkcolor=blue,
    citecolor=magenta,
    urlcolor=cyan,
    pdftitle={Keen, Kotlin Genetic Algorithms Framework},
}

% Setting up floating environments for code listings
% | SetupFloatingEnvironment - Command to set properties for floating environments like figures, tables, and listings
% | name - Name of the floating environment
\SetupFloatingEnvironment{listing}{name=Listing}

% Setting the style for code highlighting using minted package
% | usemintedstyle - Sets the style of the code highlighting
\usemintedstyle{vs}

% Basic minted configuration
% | setminted - Sets global options for minted code blocks
% | autogobble - Automatically removes common leading whitespace
% | escapeinside - Specifies delimiters to escape to LaTeX
\setminted{
  autogobble=true,
  escapeinside=~~,
  breaklines,
  breakafter={., \space},
  breakbytokenanywhere
}

% Setting the basic page style to plain
% | pagestyle - Sets the overall page style (e.g., headers and footers)
\pagestyle{plain} % this line is not necessary if previously no style has been set

% Setting up custom headers and footers using the fancyhdr package
% | fancyhead - Command to set the content of headers in the fancyhdr package
% | [LO], [RO], etc. - Specify the position of the header/footer content (e.g., Left Odd pages, Right Odd pages)
% | leftmark and rightmark - Commands that retrieve the current chapter and section names, respectively
% | thepage - Command to retrieve the current page number
\fancyhead[LO]{\leftmark}
\fancyhead[RO]{\thepage}
\fancyhead[RE]{\rightmark}
\fancyhead[LE]{\thepage}
% \fancyfoot[C]{\textit{\getvmeta{title}}}
\fancyfoot{}

% Override the plain page style with the customizations
% | fancypagestyle - Allows for customization of a specific page style
\fancypagestyle{plain}{} % this is necessary for the first page of the chapter
