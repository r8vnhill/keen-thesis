% DOCUMENT CONFIGURATION

% Remove paragraph indentation
\setlength{\parindent}{0pt}

% Add bibliography resource
\addbibresource{Keen.bib}

% Configure main document font
\setmainfont{MinionPro-Regular.otf}[
  Path = fonts/,
  BoldFont = MinionPro-Bold.otf,
  ItalicFont = MinionPro-It.otf,
  BoldItalicFont = MinionPro-BoldIt.otf,
]

% Configure monospace font (typically used for code)
\setmonofont{DejaVuSansMNerdFontMono-Regular.ttf}[
  Path = fonts/,
  BoldFont = DejaVuSansMNerdFontMono-Bold.ttf,
  ItalicFont = DejaVuSansMNerdFontMono-Oblique.ttf,
  Scale = 0.85
]

% Define hyperlink properties
\hypersetup{
    colorlinks=true,
    linkcolor=blue,
    citecolor=magenta,
    urlcolor=cyan,
    pdftitle={Keen, Kotlin Genetic Algorithms Framework},
}

% Customize floating environments for code listings
\SetupFloatingEnvironment{listing}{name=Listing}

% Define code highlighting style
\usemintedstyle{vs}

% Configure minted package
\setminted{
  autogobble=true,
  escapeinside=~~,
  breaklines,
  breakafter={., \space},
  breakbytokenanywhere
}

% Set the default page style to plain
\pagestyle{plain}

% Define custom headers and footers
\fancyhead[LO]{\leftmark}
\fancyhead[RO]{\thepage}
\fancyhead[RE]{\rightmark}
\fancyhead[LE]{\thepage}
\fancyfoot{}

% Apply customizations to the plain page style
\fancypagestyle{plain}{}

% Adjust depth of section numbering
\setcounter{secnumdepth}{3}
